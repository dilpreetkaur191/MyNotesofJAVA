# Basic linux command
ls – listing files e.g. ls -al , ls -alh
mkdir – create folder e.g. mkdir /tmp/test-folder
cd – change folder e.g. cd /tmp
rmdir – delete empty folder e.g. rmdir /tmp-folder
rm – remove file and folders with all the content e.g. rm -rf /tmp/test-folder
mv – move or rename file or folder from place to another e.g. mv /tmp/test-folder /tmp/new-test-folder
ifconfig – to check ip address (if ifconfig package is installed) e.g. ifconfig
touch – create empty file e.g. touch emptyfile.txt
df – check disk usage to know available and used space e.g. df -h
ps – check all the running processes e.g. ps -auxe , ps -aux
kill – killing process by process id from the ps command above e.g. kill -9 pid
cat – print the content of a file on the terminal or console e.g cat ENTER-FILE-NAME
